Foobar

This ruby program is an adaptation of a classic technical interview problem which displays sequences of the Foobar pattern. 

n = 0
#FOOBAR CHALLENGE
#numbers divisble by 3 = FOO
#numbers divisble by 5 = BAR
#numbers divisble by 3&5 = FOOBAR

puts "How many items do you want to see?"
items = gets.to_i

list_items = [] #empty array

(1..items).each do |x|
  if x % 3 == 0 && x % 5 == 0
    list_items << "FOOBAR"
  elsif x % 3 == 0
    list_items << "FOO"
  elsif x % 5 == 0
    list_items << "BAR"
  else
    list_items << x
  end
end

puts list_items
